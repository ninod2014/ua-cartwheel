00335                 double result = 0;

00340                 qa = computeHeading(this->getOrientation()).getComplexConjugate() * this->getOrientation();

00342 
00343                 va = this->getOrientation().getComplexConjugate().rotate(this->getVelocity());
00344                 vb = other->getOrientation().getComplexConjugate().rotate(other->getVelocity());
00345

00347                 result += 3 * (va-vb).dotProductWith(va-vb);
00348 
00349                 //and now penalize the difference in root angular velocities
00350                 va = this->getOrientation().getComplexConjugate().rotate(this->getAngularVelocity());

00352                 result += 3 * (va-vb).dotProductWith(va-vb);
