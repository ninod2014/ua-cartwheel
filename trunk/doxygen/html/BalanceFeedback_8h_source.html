00124         virtual double getFeedbackContribution(SimBiController* /* con */, CartWheel::Physics::Joint* /* j */, double /* phi */, CartWheel::Math::Vector3d d, CartWheel::Math::Vector3d v){
00125                 double dToUse = d.dotProductWith(feedbackProjectionAxis);
00126                 double vToUse = v.dotProductWith(feedbackProjectionAxis);
00127                 if (dToUse < dMin) dToUse = dMin;
00128                 if (vToUse < vMin) vToUse = vMin;
00129                 if (dToUse > dMax) dToUse = dMax;
00130                 if (vToUse > vMax) vToUse = vMax;

00132                 return dToUse * cd + vToUse * cv;
