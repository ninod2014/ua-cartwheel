00178                 double scale = 1;
00179                 //this d.z should really be d dotted with some axis - probably the same as the feedback one...
00180                 if (dTrajScale.getKnotCount() > 0)
00181                         scale *= dTrajScale.evaluate_linear(d.z);
00182 
00183                 //this v.z should really be v dotted with some axis - probably the same as the feedback one...
00184                 if (vTrajScale.getKnotCount() > 0)
00185                         scale *= vTrajScale.evaluate_linear(v.z);
00186 
00187                 if (bareTrajectory == true)
00188                         scale = 1;
00189 
00190                 if (baseTraj.getKnotCount() > 0)
00191                         baseAngle += baseTraj.evaluate_catmull_rom(phi) * scale;
00192 
00193                 if (stance == LEFT_STANCE && reverseAngleOnLeftStance)
00194                         baseAngle = -baseAngle;
00195                 if (stance == RIGHT_STANCE && reverseAngleOnRightStance)
00196                         baseAngle = -baseAngle;
00197 
00198                 double feedbackValue = computeFeedback(con, j, phi, d, v);
00199 
00200                 return CartWheel::Math::Quaternion::getRotationQuaternion(baseAngle + feedbackValue, rotationAxis);
