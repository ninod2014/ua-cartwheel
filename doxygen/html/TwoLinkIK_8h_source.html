00033                 //the solution for this comes from computation of the intersection of two circles of radii r1 and r2, located at
00034                 //p1 and p2 respectively. There are, of course, two solutions to this problem. The calling application can differentiate between these
00035                 //by passing in n or -n for the plane normal.
00036                 
00037                 //this is the distance between p1 and p2. If it is > r1+r2, then we have no solutions. To be nice about it,
00038                 //we will set r to r1+r2 - the behaviour will be to reach as much as possible, even though you don't hit the target
00039                 double r = CartWheel::Math::Vector3d(p1, p2).length();
00040                 if (r > (r1+r2) * 0.993)
00041                         r = (r1 + r2) * 0.993;
00042                 //this is the length of the vector starting at p1 and going to the midpoint between p1 and p2
00043                 double a = (r1 * r1 - r2 * r2 + r * r) / (2 * r);
00044                 double tmp = r1*r1 - a*a;
00045                 if (tmp < 0)
00046                         tmp = 0;
00047                 //and this is the distance from the midpoint of p1-p2 to the intersection point
00048                 double h = sqrt(tmp);
00049                 //now we need to get the two directions needed to reconstruct the intersection point
00050                 CartWheel::Math::Vector3d d1 = CartWheel::Math::Vector3d(p1, p2).toUnit();
00051                 CartWheel::Math::Vector3d d2 = d1.crossProductWith(n).toUnit();
00052 
00053                 //and now get the intersection point
00054                 CartWheel::Math::Point3d p = p1 + d1 * a + d2 * (-h);
00055 
00056                 return p;
00057         }
