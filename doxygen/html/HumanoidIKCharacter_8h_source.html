00030         static inline CartWheel::Math::Point3d solve(const CartWheel::Math::Point3d& p1, const CartWheel::Math::Point3d& p2, const CartWheel::Math::Vector3d& n, double r1, double r2){
00031                 //the solution for this comes from computation of the intersection of two circles of radii r1 and r2, located at
00032                 //p1 and p2 respectively. There are, of course, two solutions to this problem. The calling application can differentiate between these
00033                 //by passing in n or -n for the plane normal.
00034                 
00035                 //this is the distance between p1 and p2. If it is > r1+r2, then we have no solutions. To be nice about it,
00036                 //we will set r to r1+r2 - the behaviour will be to reach as much as possible, even though you don't hit the target
00037                 double r = CartWheel::Math::Vector3d(p1, p2).length();
00038                 if (r > r1+r2)
00039                         r = r1 + r2;
00040                 //this is the length of the vector starting at p1 and going to the midpoint between p1 and p2
00041                 double a = (r1 * r1 - r2 * r2 + r * r) / (2 * r);
00042                 //and this is the distance from the midpoint of p1-p2 to the intersection point
00043                 double h = sqrt(r1*r1 - a*a);
00044                 //now we need to get the two directions needed to reconstruct the intersection point
00045                 CartWheel::Math::Vector3d d1 = CartWheel::Math::Vector3d(p1, p2).toUnit();
00046                 CartWheel::Math::Vector3d d2 = d1.crossProductWith(n).toUnit();
00047 
00048                 //and now get the intersection point
00049                 CartWheel::Math::Point3d p = p1 + d1 * a + d2 * (-h);
00050 
00051                 return p;
00052         }

00069                 CartWheel::Math::Quaternion q;
00070 
00071                 //first off, compute the CartWheel::Math::Quaternion that rotates nLocal into nGlobal
00072                 CartWheel::Math::Vector3d axis = nLocal.crossProductWith(nGlobal);
00073                 axis.toUnit();
00074                 double ang = nLocal.angleWith(nGlobal);
00075 
00076                 q = CartWheel::Math::Quaternion::getRotationQuaternion(ang, axis);
00077 
00078                 //now q rotates nLocal into nGlobal. We now need an aditional orientation that aligns vLocal to vGlobal
00079 
00080                 //one thing to note: nLocal is perpendicular to vLocal so q*nLocal is perpendicular to q*vLocal. Also, q*nLocal is aligned to nGlobal,
00081                 //so nGlobal is perpendicular to both q*vLocal and vGlobal, which means that this rotation better be about vGlobal!!!
00082                 vLocal = q.rotate(vLocal);
00083                 axis = vLocal.crossProductWith(vGlobal);
00084                 axis.toUnit();
00085                 ang = vLocal.angleWith(vGlobal);
00086 
00087                 q = CartWheel::Math::Quaternion::getRotationQuaternion(ang, axis) * q;
00088
