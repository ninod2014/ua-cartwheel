00028 PHYSICS_DECLSPEC inline int getContactPoints(CartWheel::Math::Sphere* a, CartWheel::Math::Sphere* b, DynamicArray<ContactPoint> *cps){
00029         //compute the distance between the origins of the two CartWheel::Math::Spheres. If they are closer than a.r + b.r, then they've collided
00030         double distSquared = SQR(b->pos.x - a->pos.x) + SQR(b->pos.y - a->pos.y) + SQR(b->pos.z - a->pos.z);
00031         if (distSquared <= SQR(a->radius + b->radius)){
00032                 cps->push_back(ContactPoint());
00033                 ContactPoint* lastPoint = &(*cps)[cps->size()-1];
00034                 double dist = sqrt(distSquared);
00035                 //set the penetration depth
00036                 lastPoint->d = (a->radius + b->radius - dist);
00037                 //set the normal (pointing from b into a)
00038                 lastPoint->n.setToVectorBetween(b->pos, a->pos); lastPoint->n /= dist;
00039                 lastPoint->cp.setToOffsetFromPoint(b->pos, lastPoint->n, b->radius - lastPoint->d/2.0);
00040                 return 1;
00041         }
00042         return 0;
00043 }
